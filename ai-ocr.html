/** * TextLens™ OCR - Complete AI Post-Processing Script * Developed for AIFreePDF */ // Global state tracking let ocrFile = null; let ocrWorker = null; // Aapka Dynamic API Key Loader (Jo aap baki tools me use kar rahe hain) // Note: Is variable ko wahi name dena jo aapke config me set hai, e.g., GEMINI_API_KEY const API_KEY = "YOUR_GEMINI_API_KEY_HERE"; /** * File Loader and Preview Handler */ function loadOCRFile(event) { const file = event.target.files[0]; if (!file) return; ocrFile = file; const fileInfo = document.getElementById('ocr-file-info'); const fileName = document.getElementById('ocr-file-name'); const fileSize = document.getElementById('ocr-file-size'); const previewWrap = document.getElementById('ocr-preview-wrap'); const previewImg = document.getElementById('ocr-preview-img'); const ocrBtn = document.getElementById('ocr-btn'); const langGroup = document.getElementById('ocr-lang-group'); // UI Updates for Selected File fileName.textContent = file.name; fileSize.textContent = (file.size / (1024 * 1024)).toFixed(2) + " MB"; fileInfo.style.display = "flex"; langGroup.style.display = "block"; ocrBtn.disabled = false; // Handle Image Preview if (file.type.startsWith('image/')) { const reader = new FileReader(); reader.onload = function(e) { previewImg.src = e.target.result; previewWrap.style.display = "block"; }; reader.readAsDataURL(file); } else { // PDF handles hide image preview previewWrap.style.display = "none"; } } /** * Drag and Drop Handlers */ function handleOCRDrop(event) { event.preventDefault(); document.getElementById('ocr-zone').classList.remove('drag-over'); if (event.dataTransfer.files.length > 0) { const input = document.getElementById('ocr-input'); input.files = event.dataTransfer.files; loadOCRFile({ target: input }); } } /** * Clear Workspace State */ function clearOCRFile() { ocrFile = null; document.getElementById('ocr-input').value = ""; document.getElementById('ocr-file-info').style.display = "none"; document.getElementById('ocr-lang-group').style.display = "none"; document.getElementById('ocr-preview-wrap').style.display = "none"; document.getElementById('ocr-btn').disabled = true; document.getElementById('ocr-result').style.display = "none"; document.getElementById('ocr-loader').style.display = "none"; } /** * Core Execution: Tesseract Extraction followed by Gemini AI Clean-up */ async function runOCR() { if (!ocrFile) return; const loader = document.getElementById('ocr-loader'); const progressBar = document.getElementById('ocr-progress-bar'); const progressLabel = document.getElementById('ocr-progress-label'); const resultBox = document.getElementById('ocr-result'); const resultText = document.getElementById('ocr-result-text'); const ocrBtn = document.getElementById('ocr-btn'); // Reset UI State ocrBtn.disabled = true; resultBox.style.display = "none"; loader.style.display = "block"; progressBar.style.width = "10%"; progressLabel.innerHTML = "Initializing Tesseract Engine..."; try { const selectedLang = document.getElementById('ocr-lang').value; // 1. Initialize Client-Side Tesseract Utility const worker = await Tesseract.createWorker({ logger: m => { if (m.status === 'recognizing text') { const pct = Math.floor(m.progress * 80); // Tesseract goes up to 80% progressBar.style.width = (10 + pct) + "%"; progressLabel.innerText = `Scanning Characters (${Math.floor(m.progress * 100)}%)`; } } }); progressLabel.innerText = "Loading Language Models..."; // Multi-language split parser (eng, hin, or eng+hin) const langs = selectedLang.split('+'); for (let l of langs) { await worker.loadLanguage(l); } await worker.initialize(selectedLang); progressLabel.innerText = "Extracting Matrix Characters..."; // 2. Fetch raw textual structure from file let rawText = ""; if (ocrFile.type.startsWith('image/')) { const { data: { text } } = await worker.recognize(ocrFile); rawText = text; } else { // Processing fallback for standalone scanned PDF pages const { data: { text } } = await worker.recognize(ocrFile); rawText = text; } await worker.terminate(); // Check if raw text extraction succeeded if (!rawText || rawText.trim().length === 0) { throw new Error("Could not read any characters from this document. Please ensure the image is bright and clear."); } // 3. Switch Engine to Gemini Core Intelligence for Correction progressBar.style.width = "90%"; progressLabel.innerText = "Applying AI Text Intelligence Layer..."; // Highly targeted Multilingual Correction Engine System Prompt const systemPrompt = `You are an expert multilingual OCR text cleaner and grammar corrector. Analyze the provided raw OCR text extracted from an image or document. CRITICAL INSTRUCTIONS FOR HINDI (DEVANAGARI) & SCANNED DATA: 1. Reconstruct broken and distorted Devanagari characters, words, matras, and halants into syntactically flawless Hindi. 2. Remove all digital noise, random character alignments, and garbage symbols (e.g., replace artifacts like '#+', '|', '0', 'le', or broken numeric strings with correct context-based letters or clean standard digits). 3. If the input is mixed language (Hinglish/English + Hindi), preserve both tracks naturally while applying clean professional formatting and spacing. 4. Maintain exact line breaks and logical spacing where relevant for document layout. 5. Absolute Rule: Do NOT include any introductory greetings, meta-commentary, or wrappers like 'Here is the cleaned text:'. Directly output ONLY the final reconstructed editable text. Raw Extracted Text: """ ${rawText} """`; // Dispatch call to Google Generative Language Platform const aiResponse = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${API_KEY}`, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ contents: [{ parts: [{ text: systemPrompt }] }] }) }); if (!aiResponse.ok) { throw new Error(`Gemini Engine responded with status code: ${aiResponse.status}`); } const aiData = await aiResponse.json(); const finalCleanedText = aiData.candidates[0].content.parts[0].text; // Render response to Document Interface progressBar.style.width = "100%"; resultText.innerText = finalCleanedText.trim(); loader.style.display = "none"; resultBox.style.display = "block"; } catch (error) { console.error("OCR Optimization Failure:", error); loader.style.display = "none"; showToast(error.message || "An unexpected error occurred during processing."); } finally { ocrBtn.disabled = false; } } /** * Clipboard Operation Helper */ function copyResult(elementId) { const textElement = document.getElementById(elementId); if (!textElement) return; // Use internal text value to bypass contenteditable structural wrappers const textToCopy = textElement.innerText; navigator.clipboard.writeText(textToCopy).then(() => { showToast("📋 Text copied successfully to clipboard!"); }).catch(err => { console.error("Failed to execute copy payload:", err); showToast("❌ Unable to copy text automatically."); }); } /** * File Generation Engine (Exporting text context as client-side asset) */ function downloadOCRText() { const textElement = document.getElementById('ocr-result-text'); if (!textElement) return; const textData = textElement.innerText; if (!textData.trim()) { showToast("No content available to export."); return; } const textBlob = new Blob([textData], { type: 'text/plain;charset=utf-8' }); const downloadUrl = URL.createObjectURL(textBlob); const downloadAnchor = document.createElement('a'); downloadAnchor.href = downloadUrl; downloadAnchor.download = `SwiftMind_OCR_Export_${Date.now()}.txt`; document.body.appendChild(downloadAnchor); downloadAnchor.click(); // Memory Clean up sequence document.body.removeChild(downloadAnchor); URL.revokeObjectURL(downloadUrl); } /** * Toast Utility Notification Display */ function showToast(message) { const toast = document.getElementById('toast'); if (!toast) return; toast.textContent = message; toast.classList.add('show'); setTimeout(() => { toast.classList.remove('show'); }, 3500); }
